     4  module formater(
     5                   input                      clk_i,
     6                   input                      rstn_i,
     7  					  
     8                   //connect with arbiter
     9  				         output                     f2a_ack_o,
     10                   output                     fmt_id_req_o,
     11                   input                      a2f_val_i,
     12                   input    [1:0]             a2f_id_i,
     13                   input    [31:0]            a2f_data_i,
     14  					       input 	  [2:0] 				  pkglen_sel_i,
     15  					  
     16                  //connect with outside
     17  				        input                      fmt_grant_i,
     18                  output   [1:0]             fmt_chid_o,                  
     19                  output   [5:0]             fmt_length_o,                  
     20                  output                     fmt_req_o,
     21                  output   [31:0]            fmt_data_o,
     22                  output                     fmt_start_o,
     23                  output                     fmt_end_o
     24                 );
     25  
     26  reg [5:0]     length_r;
     27  reg [31:0]    fmt_fifo [0:31];
     28  reg [5:0]     cnt_rec_r;
     29  reg [5:0]     cnt_sen_r;
     30  reg [31:0]    slv0_buffer_r;
     31  reg [31:0]    slv1_buffer_r;
     32  reg [31:0]    slv2_buffer_r;
     33  reg [31:0]    fmt_data_r;
     34  reg [2:0]     c_state, n_state;
     35  
     36  reg buffer0_val_r;
     37  reg buffer1_val_r;
     38  reg buffer2_val_r;
     39  
     40  reg fmt_end_r;
     41  reg fmt_start_r;
     42  reg fmt_req_r;
     43  reg fmt_ack_r;
     44  reg fmt_send_r;
     45  reg fmt_id_req_r;
     46  
     47  
     48  //---------------------------------package length decode-------------------------// 
     49  always @ (*)   						
     50  begin
     51    if (!rstn_i) length_r = 6'd32;
     52    else 
     53    case (pkglen_sel_i) 
     54      3'd0 : length_r = 6'd4;
     55      3'd1 : length_r = 6'd8;
     56      3'd2 : length_r = 6'd16;
     57      3'd3 : length_r = 6'd32;
     58      default : length_r = 6'd32;
     59    endcase
     60  end 
     61  
     62  //-----------------------------formater fifo write pointer increase---------------------//
     63  always @ (posedge clk_i or negedge rstn_i)
     64  begin
     65    if (!rstn_i) cnt_rec_r <= 1'b0;
     66    else if (fmt_id_req_r) cnt_rec_r <= 1'b0;
     67    else 
     68    case (a2f_id_i)
     69      2'b00 : if ((a2f_val_i | buffer0_val_r) && fmt_ack_r) cnt_rec_r <= cnt_rec_r + 1'b1;
     70      2'b01 : if ((a2f_val_i | buffer1_val_r) && fmt_ack_r) cnt_rec_r <= cnt_rec_r + 1'b1;
     71      2'b10 : if ((a2f_val_i | buffer2_val_r) && fmt_ack_r) cnt_rec_r <= cnt_rec_r + 1'b1;
     72      default : cnt_rec_r <= cnt_rec_r;
     73    endcase 
     74  end 
     75  
     76  //--------------------------fullfill formater fifo && buffer------------------------------------//
     77  always @ (posedge clk_i or negedge rstn_i)
     78  begin
     79    if (!rstn_i) begin  fmt_fifo[cnt_rec_r] <= 32'hffff_ffff;
     80                        slv0_buffer_r <= 32'hffff_ffff; 
     81  					            buffer0_val_r <= 1'b0;
     82                        slv1_buffer_r <= 32'hffff_ffff;
     83  					            buffer1_val_r <= 1'b0;
     84                        slv2_buffer_r <= 32'hffff_ffff;
     85  					            buffer2_val_r <= 1'b0;
     86                  end 
     87    else if (fmt_ack_r) 
     88          case (a2f_id_i)
     89                2'b00 : if (buffer0_val_r && fmt_ack_r) begin fmt_fifo[cnt_rec_r] <= slv0_buffer_r; buffer0_val_r <= 1'b0; end 
     90                else if (a2f_val_i && fmt_ack_r) fmt_fifo[cnt_rec_r] <= a2f_data_i;
     91                2'b01 : if (buffer1_val_r && fmt_ack_r) begin fmt_fifo[cnt_rec_r] <= slv1_buffer_r; buffer1_val_r <= 1'b0; end 
     92                else if (a2f_val_i && fmt_ack_r) fmt_fifo[cnt_rec_r] <= a2f_data_i;
     93                2'b10 : if (buffer2_val_r && fmt_ack_r) begin fmt_fifo[cnt_rec_r] <= slv2_buffer_r; buffer2_val_r <= 1'b0; end 
     94                else if (a2f_val_i && fmt_ack_r) fmt_fifo[cnt_rec_r] <= a2f_data_i;
     95          endcase
     96      else 
     97        case (a2f_id_i)
     98              2'b00 : if (a2f_val_i && !fmt_ack_r) begin slv0_buffer_r <= a2f_data_i; buffer0_val_r <= 1'b1; end
     99              2'b01 : if (a2f_val_i && !fmt_ack_r) begin slv1_buffer_r <= a2f_data_i; buffer1_val_r <= 1'b1; end
     100              2'b10 : if (a2f_val_i && !fmt_ack_r) begin slv2_buffer_r <= a2f_data_i; buffer2_val_r <= 1'b1; end 
     101        endcase 
     102  
     103  end 
     104  
     105  //----------------------------formater fifo read pointer increase----------------------//
     106  always @ (posedge clk_i or negedge rstn_i)
     107  begin 
     108    if (!rstn_i) cnt_sen_r <= 1'b0;
     109    else if (fmt_id_req_r) cnt_sen_r <= 1'b0;
     110    else if (fmt_send_r) cnt_sen_r <= cnt_sen_r + 1'b1;
     111  end 
     112  
     113  //--------------------------empty formater fifo-------------------------------------//
     114  always @ (*)
     115  begin
     116  	if (!rstn_i) fmt_data_r <= 32'hffff_ffff;
     117  	else if (fmt_send_r) fmt_data_r <= fmt_fifo[cnt_sen_r];
     118  	else fmt_data_r <= 32'hffff_ffff;
     119  end 
     120  
     121  
     122  
     123  //-------------------------------------FSM start ------------------------------//
     124  parameter FMT_REQ = 3'b000;
     125  parameter FMT_WAIT_GRANT = 3'b001;
     126  parameter FMT_START = 3'b011;
     127  parameter FMT_SEND = 3'b010;
     128  parameter FMT_END = 3'b110;
     129  parameter FMT_IDLE = 3'b111;
     130  
     131  always @ (posedge clk_i or negedge rstn_i)
     132  if (!rstn_i) c_state <= FMT_IDLE;
     133  else c_state <= n_state;
     134  
     135  always @ (*)                      //need length_r
     136  begin
     137  	if (!rstn_i) 
     138  	n_state = FMT_IDLE;
     139  	else case (c_state)
     140  	FMT_IDLE : 	begin
     141  						if (cnt_rec_r == length_r - 2'd1) //need to decode and pay attention to cnt_rec_r start from 0 
     142  							n_state = FMT_REQ;      //pay attention to datapath and FSM structure
     143  						else 
     144  							n_state = FMT_IDLE;
     145  					end 
     146  					
     147  	FMT_REQ : if(cnt_rec_r >= length_r)	n_state = FMT_WAIT_GRANT;
     148              else n_state = FMT_REQ;
     149  	
     150  	FMT_WAIT_GRANT : 	begin
     151  								if(fmt_grant_i) n_state = FMT_START;
     152  								else n_state = FMT_WAIT_GRANT;
     153  							end 
     154  							
     155  	FMT_START :  n_state = FMT_SEND ;
     156  	
     157  	FMT_SEND  : 	begin
     158  							if (cnt_sen_r == length_r - 2'd2) n_state = FMT_END;
     159  							else n_state = FMT_SEND;
     160  						end 
     161  						
     162  	FMT_END : n_state = FMT_IDLE;	
     163  	
     164  	default : n_state = FMT_IDLE;
     165  	endcase 
     166  end 
     167  
     168  always @ (*)
     169  begin
     170    if (!rstn_i)begin
     171                  fmt_end_r = 1'b0;
     172                  fmt_start_r = 1'b0;
     173                  fmt_req_r = 1'b0;
     174                  fmt_ack_r = 1'b0;
     175                  fmt_send_r = 1'b0;
     176                  fmt_id_req_r = 1'b1;
     177                end
     178    else case (c_state)
     179      FMT_IDLE : begin
     180        if (a2f_id_i != 2'b11) fmt_ack_r = 1'b1;
     181        else fmt_ack_r = 1'b0;
     182        if (a2f_id_i == 2'b11) fmt_id_req_r = 1'b1;
     183        else fmt_id_req_r = 1'b0;
     184        fmt_end_r = 1'b0;
     185        fmt_start_r = 1'b0;
     186        fmt_req_r = 1'b0;
     187        fmt_send_r = 1'b0;
     188      end 
     189  
     190      FMT_REQ : begin
     191        if (cnt_rec_r >= length_r)  fmt_ack_r = 1'b0;
     192        else fmt_ack_r = 1'b1;
     193        fmt_req_r = 1'b1;
     194        //fmt_ack_r = 1'b0;
     195        fmt_end_r = 1'b0;
     196        fmt_start_r = 1'b0;
     197        fmt_send_r = 1'b0;
     198        fmt_id_req_r = 1'b0;
     199      end
     200  
     201      FMT_WAIT_GRANT : begin
     202        fmt_req_r = 1'b1;
     203        fmt_ack_r = 1'b0;
     204        fmt_start_r = 1'b0;
     205        fmt_end_r = 1'b0;
     206        fmt_send_r = 1'b0;
     207        fmt_id_req_r = 1'b0;
     208      end
     209  
     210      FMT_START : begin
     211        fmt_req_r = 1'b0;
     212        fmt_ack_r = 1'b0;
     213        fmt_start_r = 1'b1;
     214        fmt_end_r = 1'b0;
     215        fmt_send_r = 1'b1;
     216        fmt_id_req_r = 1'b0;
     217      end
     218  
     219      FMT_SEND : begin
     220        fmt_req_r = 1'b0;
     221        fmt_ack_r = 1'b0;
     222        fmt_start_r = 1'b0;
     223        fmt_end_r = 1'b0;
     224        fmt_send_r = 1'b1;
     225        fmt_id_req_r = 1'b0;
     226      end
     227  
     228      FMT_END : begin 
     229        fmt_req_r = 1'b0;
     230        fmt_ack_r = 1'b0;
     231        fmt_start_r = 1'b0;
     232        fmt_end_r = 1'b1;
     233        fmt_send_r = 1'b1;
     234        fmt_id_req_r = 1'b1;
     235      end
     236    endcase 
     237        
     238  end //end of FSM
     239  
     240   
     241  assign fmt_id_req_o = fmt_id_req_r;
     242  assign fmt_chid_o = a2f_id_i;
     243  assign fmt_length_o = length_r;
     244  assign fmt_req_o = fmt_req_r;
     245  assign fmt_start_o = fmt_start_r;
     246  assign fmt_end_o = fmt_end_r; 
     247  assign f2a_ack_o = fmt_ack_r;
     248  assign fmt_data_o = fmt_data_r;
     249  
     250  endmodule
